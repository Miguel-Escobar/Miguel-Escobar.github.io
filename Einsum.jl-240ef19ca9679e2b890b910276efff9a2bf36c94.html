95 # don't need to check rhs, already done above

141 "To use ", expr.head, " try @einsum instead.")))

143 end

243 Construct a nested loop around `expr`, using `indices` in ranges `axis_exprs`.

256 function nest_loops(expr::Expr,
257 index_names::Vector{Symbol}, axis_expressions::Vector{Expr},
258 simd::Bool, threads::Bool)
259 isempty(index_names) && return expr
260
261 # Add @simd to the innermost loop, if required
262 # and @threads to the outermost loop
263 expr = nest_loop(expr, index_names[1], axis_expressions[1],
264 simd, threads && length(index_names) == 1)
265
266 # Add remaining for loops
267 for j = 2:length(index_names)
268 expr = nest_loop(expr, index_names[j], axis_expressions[j],
269 false, threads && length(index_names) == j)
270 end
271
272 return expr
273 end

321 function extractindices!(expr::Expr,
322 array_names::Vector{Symbol},
323 index_names::Vector{Symbol},
324 axis_expressions::Vector{Expr})
325 if Meta.isexpr(expr, :ref) # e.g. A[i,j,k]
326 array_name = expr.args[1]
327 push!(array_names, array_name)
328
329 # expr.args[2:end] are indices (e.g. [i,j,k])
330 for (dimension, index_expr) in enumerate(expr.args[2:end])
331 pushindex!(index_expr, array_name, dimension,
332 array_names, index_names, axis_expressions)
333 end
334 elseif Meta.isexpr(expr, :call)
335 # e.g. 2*A[i,j], transpose(A[i,j]), or A[i] + B[j], so
336 # expr.args[2:end] recursively contain the individual tensor
337 # expressions (e.g. [A[i], B[j]])
338 for arg in expr.args[2:end]
339 extractindices!(arg, array_names, index_names, axis_expressions)
340 end
341 else
342 throw(ArgumentError("Invalid expression head: `:$(expr.head)`"))
343 end
344
345 return return array_names, index_names, axis_expressions
346 end

360 function pushindex!(expr::Expr, array_name::Symbol, dimension::Int,
361 array_names, index_names, axis_expressions)
362 # e.g. A[i+:offset] or A[i+5]
363 # expr is an Expr in this case
364 # We restrict it to be a Symbol (e.g. :i) followed by either
365 # a number or quoted expression.
366 # As before, push :i to index list
367 # Need to add/subtract off the offset to dimension list
368
369 if Meta.isexpr(expr, :call) && length(expr.args) == 3
370 op = expr.args[1]
371
372 index_name = expr.args[2]
373 @assert typeof(index_name) == Symbol
374
375 offset_expr = expr.args[3]
376
377 if offset_expr isa Integer
378 offset = expr.args[3]::Integer
379 elseif offset_expr isa Expr && Meta.isexpr(offset_expr, :quote)
380 offset = offset_expr.args[1]
381 elseif offset_expr isa QuoteNode
382 offset = offset_expr.value
383 else
384 throw(ArgumentError("Improper expression inside reference on rhs"))
385 end
386
387 # push :i to indices we're iterating over
388 push!(index_names, index_name)
389
390 # need to invert + or - to determine iteration range
391 if op == :+
392 push!(axis_expressions, :(size($array_name, $dimension) - $offset))
393 elseif op == :-
394 push!(axis_expressions, :(size($array_name, $dimension) + $offset))
395 else
396 throw(ArgumentError(string("Operations inside ref on rhs are ",
397 "limited to `+` and `-`")))
398 end
399 elseif Meta.isexpr(expr, :quote)
400 return nothing
401 else
402 throw(ArgumentError("Invalid index expression: `$(expr)`"))
403 end
