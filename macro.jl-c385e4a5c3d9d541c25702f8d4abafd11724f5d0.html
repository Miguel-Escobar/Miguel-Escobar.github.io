4 """
5 @tullio C[i,k] := A[i,j] * B[j,k]
6 @tullio F[i,k] := \$α * D[i].field[j] * E[col=k, row=j] + \$β
7
8 This is a replacement for `@einsum` which understands a bit more syntax.
9 The expression on the right is summed over all possible valued of the free index `k`,
10 and `:=` makes a new array `C`, while `=` and `+=` would write into an existing one.
11 Scalar arguments should have a dollar sign, like `\$α` or `A[i,\$γ]`.
12
13 @tullio G[i,j] := M[i+x+1, j+y+1] * K[x,y]
14 @tullio H[i,j] := M[2i+x, 2j+y] (x in -1:1, y in -1:1)
15
16 Shifts and scaling of indices are allowed, including shifts by other indices.
17 Ranges can be provided as shown, for under-constrained indices.
18 If they are over-constrained, shifted indices run over the intersection allowed by all constraints,
19 while un-shifted indices demand agreement between them (e.g. `axes(A,2) == axes(B,1)` above).
20
21 @tullio (*) L[i] := A[J[k]+2, i] / B[k]^2
22
23 This is a product instead of a sum, which could also enabled by writing `L[i] *= ...` (in-place).
24 You can use any reduction function such as `@tullio (max) M[i,j] := ...`.
25 Indexing by `J[k]+2` here demands `issubset(J, axes(A,1) .- 2)`.
26
27 @tullio N[j] := sqrt <| M[i,j]^2
28
29 Pipe operators `|>` and `<|` apply a function after the sum, here `N ≈ map(norm, eachcol(M))`.
30 Underscores create functions, e.g. `|> sqrt(_ / V[i])` where clearly `i` must not have been summed.
31
32 See the readme for further options.
33 """

115 function parse_options(exs...)

132 # Actual options:
133 if isexpr(ex, :(=)) && haskey(OPTS, ex.args[1])
134 checklegal(ex.args[1], ex.args[2])
135 opts[ex.args[1]] = ex.args[2]
136
137 # Init & pad keyword
138 elseif isexpr(ex, :(=)) && ex.args[1] == :init
139 opts[:init] = ex.args[2]
140 elseif isexpr(ex, :(=)) && ex.args[1] == :pad
141 opts[:pad] = ex.args[2]
142
143 # Nograd keyword
144 elseif isexpr(ex, :(=)) && ex.args[1] == :nograd

151 end
152
153 # Ranges specified outside:

156 elseif isexpr(ex, :tuple) && isexpr(ex.args[1], :call) && ex.args[1].args[1] in [:in, :∈]

162 # Reduction function
163 elseif ex isa Symbol
164 opts[:redfun] = ex

166 # The main course!
167 elseif ex isa Expr

169 expr = ex
170 else

174 if isnothing(expr) # if run with no expression, it updates global options
175 _VERBOSE[] = opts[:verbose]

177 _THREADS[] = opts[:threads]

224 function parse_input(expr, store)
225
226 # Equals sign & friends:
227 if @capture_(expr, left_ := right_ )
228 store.newarray = true
229 elseif @capture_(expr, left_ = right_ )
230 elseif @capture_(expr, left_ += right_ )
231 store.plusequals = true
232 store.redfun == :+ || throw("can't use += with reduction $(store.redfun)")
233 elseif @capture_(expr, left_ *= right_ )
234 store.plusequals = true # slightly abusing the name of the flag!
235 if store.redfun == :+ # default, then we change it?
236 store.verbose>0 && @info "inferring reduction by *, because of lhs *= rhs"
237 store.redfun = :*
238 elseif store.redfun == :*
239 else
240 throw("can't use *= with reduction $(store.redfun)")
241 end
242 elseif @capture_(expr, left_ ^= right_ )
243 store.redfun == :+ && throw("can't use ^= with reduction +, please use +=")
244 store.redfun == :* && throw("can't use ^= with reduction *, please use *=")
245 store.plusequals = true
246 else
247 throw("can't understand input, expected A[] := B[] (or with =, or +=, *=, ^=) got $expr")
248 end
249
250 # Left hand side:
251 if @capture_(left, Z_[leftraw__] )
252 elseif @capture_(left, [leftraw__] )
253 Base.depwarn("to omit a name for the output, please write `_[i,j] := ...` with an underscore (for Tullio ≥ 0.2.14)", Symbol("@tullio"))
254 Z = :_
255 elseif left isa Symbol # complete reduction
256 store.newarray = true
257 store.scalar = true
258 store.leftscalar = left # because store.leftarray will be the array
259 leftraw = Any[1,] # the gradient still indexes a fake 1D array
260 expr.head == :(+=) && push!(store.scalars, left)
261 Z = ZED
262 else
263 throw("can't understand LHS, expected A[i,j,k], got $left")
264 end
265 leftraw2 = tidyleftraw(leftraw, store)
266 store.leftind = filter(i -> i isa Symbol && !is_const(i), leftraw2) # this gives correct outer loop order
267
268 if Z == :_
269 store.newarray || throw("can't write into an array whose name isn't given!")
270 Z = ZED
271 end
272 store.leftarray = Z
273
274 store.leftraw = finishleftraw(leftraw2, store)
275 if store.newarray && !allunique(store.leftind)
276 store.zero = true # making diagonals, etc.
277 end
278 if !(store.newarray)
279 saveconstraints(Z, leftraw, store, false) # this adds to leftind, e.g. A[2i+1] = ..., is that bad??
280 if store.plusequals # A[J[k]] += is unsafe, A[J[k]] = is not.
281 detectunsafe(left, store.unsafeleft, store)
282 store.unsafeleft = setdiff(store.unsafeleft, store.leftraw) # and A[J[k],k] += ... is safe.
283 end
284 end
285
286 # Right hand side
287 detectunsafe(right, store.unsaferight, store)
288 right2 = MacroTools_prewalk(rightwalk(store), right)
289
290 if isexpr(right2, :call) && right2.args[1] in (:|>, :<|)
291 if right2.args[1] == :|>
292 store.finaliser = makefinaliser(right2.args[3], store)
293 store.right = MacroTools_postwalk(dollarwalk(store), right2.args[2])
294 elseif right.args[1] == :<|
295 store.finaliser = makefinaliser(right2.args[2], store)
296 store.right = MacroTools_postwalk(dollarwalk(store), right2.args[3])
297 end
298 if store.scalar
299 throw("can't use a finaliser $(right2.args[1]) with scalar output")
300 end
301 else
302 store.right = MacroTools_postwalk(dollarwalk(store), right2)
303 store.finaliser = :identity
304 end

306 unique!(store.scalars)
307 unique!(store.arrays)
308 unique!(store.leftind)
309 store.sharedind = unique!(setdiff(store.sharedind, store.notfree))
310 store.rightind = unique!(setdiff(store.rightind, store.notfree))
311 store.unsaferight = union(setdiff(store.unsaferight, store.sharedind), store.shiftedind)
312 any(==(:_), vcat(store.leftind, store.rightind)) && throw("can't use _ as an index name")
313
314 unique!(store.outpre) # kill mutiple assertions, and evaluate any f(A) only once
315
316 if store.newarray && Z in store.arrays
317 throw("can't create a new array $Z when this also appears on the right")

626 function parse_ranges(ranges, store) # now runs after parse_input
627 for (i,r) in ranges
628 if isexpr(i, Symbol("'")) # catch primes!
629 i = Symbol(i.args[1], "′")
630 end
631 push!(store.rightind, i)
632 v = get!(store.constraints, i, [])
633 if isexpr(r, :call) && r.args[1] == :(:) && length(r.args) == 3
634 # for a literal range, write OneTo(10) or 0:9 directly into constraints
635 if r.args[2] == 1 && r.args[3] isa Integer
636 push!(v, :(Base.OneTo($(r.args[3]))))
637 continue
638 elseif r.args[2] isa Integer && r.args[3] isa Integer
639 push!(v, r)
640 continue
641 end
642 end
643 # for axes(A,2) where A is already available, just save it
644 if isexpr(r, :call) && r.args[1] in (:axes, :eachindex) && r.args[2] in store.arrays
645 push!(v, r)
646 continue
647 end
648 # for anything else, treat it as a scalar argument
649 if r isa Symbol
650 push!(store.scalars, r)
651 push!(v, r)
652 else
653 s = Symbol(string("≪", r, "≫"))
654 push!(store.outpre, :(local $s = $r))
655 str = "expected a range for ($i in $r), got "
656 push!(store.outpre, :($s isa AbstractRange || throw($str * string($r))))
657 push!(store.scalars, s)
658 push!(v, s)
659 end
660 end
661 unique!(store.rightind)
662 unique!(store.scalars)
663 store.redind = setdiff(store.rightind, store.leftind)
664 end

668 function index_ranges(store)
669
670 todo = Set(vcat(store.leftind, store.redind))
671 done = Dict{Symbol,Any}()
672
673 for (i,j,r_i,r_j) in store.constraintpairs
674
675 if isnothing(i) # case of A[j + I[k]]
676 v = get!(store.constraints, j, [])
677 push!(v, r_j)
678 elseif isnothing(j)
679 v = get!(store.constraints, i, [])
680 push!(v, r_i)
681
682 elseif haskey(store.constraints, i) && i in todo
683 resolveintersect(i, store, done) # use existing knowledge to fix i's range
684 pop!(todo, i)
685 v = get!(store.constraints, j, []) # and then allow j's range to depend on that
686 push!(v, r_j)
687 elseif haskey(store.constraints, j) && j in todo
688 resolveintersect(j, store, done)
689 pop!(todo, j)
690 v = get!(store.constraints, i, [])
691 push!(v, r_i)
692 end
693 end
694
695 for i in todo
696 haskey(store.constraints, i) || throw("unable to infer range of index $i")
697 if i in store.shiftedind
698 resolveintersect(i, store, done)
699 else
700 resolvestrict(i, store, done)
701 end
702 deli = Symbol(DEL,i)
703 if deli in store.scalars # magic shift on LHS

706 :(local $axi_del = Base.OneTo($length($axi))))
707 # You can't compute deli inside Act! as doesn't always see full range of i.
708 # But if you make it a scalar argument, then it's an argument of Make, hence
709 push!(store.outpre, :(local $deli = 0)) # ... this awful hack.
710 end
711 end
712
713 append!(store.outex, store.axisdefs)
714
715 if store.verbose > 0
716 if !isempty(store.leftind)
717 lex = map(i -> Expr(:(=), i, done[i]), store.leftind)
718 push!(store.outex, :(@info "left index ranges" $(lex...)))

720 if !isempty(store.redind)
721 rex = map(i -> Expr(:(=), i, done[i]), store.redind)
722 push!(store.outex, :(@info "reduction index ranges" $(rex...)))
723 end
724 end
725 end

727 resolvestrict(i, store, done) = begin
728 res = first(store.constraints[i])
729 ax_i = Symbol(AXIS, i)
730 push!(store.axisdefs, :( local $ax_i = $res ))
731 done[i] = res
732 for alt in store.constraints[i][2:end] # in which case it shouldn't be a Set
733 str = "range of index $i must agree"
734 push!(store.axisdefs, :( $alt == $res || $throw($str) ))
735 end
736 end
737
738 resolveintersect(i, store, done) = begin
739 res = if isempty(store.constraints[i])
740 throw("unable to infer range of index $i")
741 elseif length(store.constraints[i])==1

747 push!(store.axisdefs, :( local $ax_i = $res ))
